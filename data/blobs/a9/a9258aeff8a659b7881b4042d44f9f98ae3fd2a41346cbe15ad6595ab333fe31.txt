package com.pratice.first;

import java.util.stream.IntStream;

public class Sum {
	
	public static void main(String[] args) {
		
		int sum=IntStream.rangeClosed(1, 10).filter(Sum::isPrime).sum();
		System.out.println(sum);
		
	}

	public static boolean isPrime(int n) {
		int f=0,i;
		for(i=2;i<=n/2;i++) {
			if(n%i==0) {
				f=1;
				break;
			}
		}
		if(f==0 && n!=1 && n!=0) {
			return true;
		}else {
			return false;
		}
	}
	
}

import Interfaces.Move;
import Interfaces.Speak;
import Interfaces.Swimmable;

public class Dog implements Runnable, Swimmable, Speak {
        public String name;
        public int age;

    public Dog(String name, int age) {
        this.name = name;
        this.age = age;
    }


    @Override
    public void run() {
        System.out.println("Собака бежит");
    }

    @Override
    public void speak() {
        System.out.println("Гав гав");
    }

    @Override
    public void swim() {
        System.out.println("Собака плывет");
    }
}

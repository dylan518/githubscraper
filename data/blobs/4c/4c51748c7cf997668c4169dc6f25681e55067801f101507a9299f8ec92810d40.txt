    private PartyMessageScreen partyMessageScreen;

    private HashMap<Integer, Boolean> messageType;

    {

        modEventBus.addListener(this::commonSetup);

        // Register the Deferred Register to the mod event bus so blocks get registered
        BLOCKS.register(modEventBus);
        // Register the Deferred Register to the mod event bus so items get registered
        ITEMS.register(modEventBus);
        // Register the Deferred Register to the mod event bus so tabs get registered
        CREATIVE_MODE_TABS.register(modEventBus);

        modContainer.registerConfig(ModConfig.Type.COMMON, Config.SPEC);

        initPartyMessageScreen();
    }

    private void initPartyMessageScreen() {
        this.partyMessageScreen = new PartyMessageScreen(Component.literal("Party Chat Manager"));
        // This will load the saved state from JSON, including any selected party
        this.siblingLoc = partyMessageScreen.getSiblingLoc();
    }

    private void commonSetup(final FMLCommonSetupEvent event)
    {

        LOGGER.info("HELLO FROM COMMON SETUP");

        if (Config.logDirtBlock)
            LOGGER.info("DIRT BLOCK >> {}", BuiltInRegistries.BLOCK.getKey(Blocks.DIRT));

        LOGGER.info(Config.magicNumberIntroduction + Config.magicNumber);

        Config.items.forEach((item) -> LOGGER.info("ITEM >> {}", item.toString()));
    }

    public void sendMessage(String message) {
        if(Minecraft.getInstance().player != null && debug) {
            Objects.requireNonNull(Minecraft.getInstance().player).sendSystemMessage(Component.literal(message));
        }
    }

    // You can use SubscribeEvent and let the Event Bus discover methods to call
    @SubscribeEvent
    public void onServerStarting(ServerStartingEvent event)
    {
        // Do something when the server starts
        LOGGER.info("HELLO from server starting");
    }

    @SubscribeEvent
    public void onClientTick(ClientTickEvent.Post event) {
        while (EXAMPLE_MAPPING.get().consumeClick()) {
            if (Minecraft.getInstance().player != null) {
                // Use existing instance instead of creating new one
                Minecraft.getInstance().setScreen(this.partyMessageScreen);
            }
        }
    }

    @SubscribeEvent
    public void onChatMessage(ClientChatReceivedEvent event) {
        if (Minecraft.getInstance().player == null || Minecraft.getInstance().screen instanceof PartyMessageScreen) {
            sendMessage("ln 180");
            return;
        }
        sendMessage(String.valueOf(siblingLoc));
        if (partyMessageScreen.getSiblingLoc() == -1){
            if(userMessaged) {
                sendMessage("User has sent message");
                fetchSiblingLoc(event);
                userMessaged = false;
            }
            sendMessage(String.valueOf(siblingLoc));
            //This means that the player has yet to select a user message. We need them to do so, so that we can scrape the message for the username location
            /*Style clickStyle = Style.EMPTY
                    .withColor(ChatFormatting.BLUE)
                    .withUnderlined(true)
                    .withHoverEvent(new HoverEvent(HoverEvent.Action.SHOW_TEXT,
                            Component.literal("Click to analyze this message")));

            MutableComponent clickableText = Component.literal(" [Click me if this is a message from a user]")
                    .withStyle(clickStyle);

            event.getMessage().getSiblings().add(clickableText);*/
            chatReceivedEvent = event;
            return;
        }
        if (partyMessageScreen != null) {
            sendMessage("ln202");
            String currentParty = partyMessageScreen.getSelectedParty();
            if (currentParty == null || partyMessageScreen.isBypass()) {
                String form = String.format("Current Party: %s, is bypassed: %b", currentParty, partyMessageScreen.isBypass());
                sendMessage("ln205");
                sendMessage(form);
                return;
            }

            partyMembers = partyMessageScreen.getSelectedPartyMembers();
            siblings = event.getMessage().getSiblings();

            LOGGER.info("Siblings Size: {}", siblings.size());
            for (int i = 0; i < siblings.size(); i++) {
                LOGGER.info("Sibling {}: {}", i, siblings.get(i));
            }

            componentLength = siblings.size();
        }
        sendMessage("ln219");
        if(siblingLoc == clientChatSibLoc){
            sendMessage("ln220");
            String fs = String.format("Username: %s, Message author: %s",Minecraft.getInstance().player.getUUID(),event.getSender());
            sendMessage(fs);
            //TODO handle client message, ignore siblingLoc as it is a client message and not a system message. It will have a sender paired with the event so scraping message for username is redundant
            if(!isInParty) {
                isInParty = partyMembers.stream()
                        .anyMatch(member -> event.getSender().toString().contains(member.getUuid().toString()));
            }
            if(!isUser) {
                isUser = event.getSender().toString().contains(String.valueOf(Minecraft.getInstance().player.getUUID()));
            }
        }else{
            sendMessage("ln230");
            //TODO get the sibling at siblingLoc and check both literal as well as hover for username/playername/name/IGN
            Component sibling;
            if(siblings.size() >= siblingLoc){
                sibling = siblings.get(siblingLoc);
            }else{
                return;
            }
            HoverEvent hoverEvent = sibling.getStyle().getHoverEvent();
            String playerName = "";
            try {
                if (hoverEvent != null) {
                    playerName = Objects.requireNonNull(hoverEvent.getValue(HoverEvent.Action.SHOW_TEXT)).getString();
                }
            } catch (Exception e) {
                LOGGER.error("Error while getting hover event", e);
            }
            String finalPlayerName = playerName;
            if(!isInParty) {
                isInParty = partyMembers.stream()
                        .anyMatch(member -> sibling.getContents().toString().contains(member.getName())) ||
                            partyMembers.stream()
                        .anyMatch(member -> finalPlayerName.contains(member.getName()));
            }
            if(!isUser) {
                isUser = sibling.getContents().toString().contains(Minecraft.getInstance().player.getName().getString()) ||
                         playerName.contains(Minecraft.getInstance().player.getName().getString());
            }
        }
        String p = String.format("Is in party: %b, is a user message: %b, is user: %b", isInParty, isUserMessage, isUser);
        sendMessage(p);
        LOGGER.info("isUserMessage: {}", isUserMessage);
        if ((isInParty) || isUser) {
            sendMessage("User is either in party or current player");
        }else{
            sendMessage("Not in party");
            LOGGER.info("User is not a party member");
            event.setCanceled(true);
        }
        isUserMessage = false;
        isInParty = false;
        isUser = false;
    }
    @SubscribeEvent
    public void onChat(ClientChatEvent event) {
        userMessaged = true;
    }

    private void fetchSiblingLoc(ClientChatReceivedEvent event) {
        //TODO Scrape through the chat event and determine the index of siblings as well as if it is a hovered event that contains the username
        onlinePlayers = Objects.requireNonNull(Minecraft.getInstance().getConnection()).getOnlinePlayers();
        if (event.isSystem()){
            //TODO handle system message
            for(int index = 0; index < event.getMessage().getSiblings().size(); index++){
                Component sibling = event.getMessage().getSiblings().get(index);
                HoverEvent hoverEvent = sibling.getStyle().getHoverEvent();
                String playerName = "";
                if (hoverEvent != null) {
                    playerName = Objects.requireNonNull(hoverEvent.getValue(HoverEvent.Action.SHOW_TEXT)).getString();
                }
                for (PlayerInfo onlinePlayer : onlinePlayers) {
                    /*siblings.get(i).getString().contains(onlinePlayer.getProfile().getName())*/
                    try {
                        if(sibling.getString().contains(onlinePlayer.getProfile().getName()) || Objects.requireNonNull(Objects.requireNonNull(sibling.getStyle().getHoverEvent()).getValue(HoverEvent.Action.SHOW_TEXT)).getString().contains(Objects.requireNonNull(Minecraft.getInstance().player).getName().getString())){
                            //Literal component or hover event contains username, save location
                            sendMessage(sibling.getString());
                            this.siblingLoc = index;
                            partyMessageScreen.setSiblingLoc(index);
                            partyMessageScreen.saveChanges();
                            sendMessage("Found username: "+sibling.getContents()+", or "+ Objects.requireNonNull(sibling.getStyle().getHoverEvent()).getValue(HoverEvent.Action.SHOW_TEXT)+" in chat");
                            return;
                        }else{
                            //No username found in literal component, so reset siblingLoc
                            sendMessage("Username not found in chat");
                            LOGGER.error("Username not found in message");
                            partyMessageScreen.setSiblingLoc(-1);
                            this.siblingLoc = -1;
                            userMessaged = false;
                        }
                    }catch (Exception e){
                        LOGGER.error("Error while fetching sibling loc", e);
                        sendMessage("Error while fetching sibling loc: " + e.getMessage());
                    }

                }
            }
            partyMessageScreen.saveChanges();

        }else{
            //TODO handle user message
            sendMessage("Message is clientside so: " + clientChatSibLoc);
            this.siblingLoc = clientChatSibLoc;
            partyMessageScreen.setSiblingLoc(clientChatSibLoc);
        }
    }
